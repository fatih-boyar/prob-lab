from shiny import App, ui, render
import numpy as np
import matplotlib.pyplot as plt

import nest_asyncio
nest_asyncio.apply()

# 1. UI: Inputs on TOP to save vertical space
app_ui = ui.page_fluid(
    ui.card(
        ui.layout_columns(
            ui.input_slider("true_p", "True Preference", 0.0, 1.0, 0.5, step=0.01),
            ui.input_slider("sample_size", "Sample Size", 10, 500, 50),
            ui.input_radio_buttons("alpha", "Significance", 
                                   {"0.01": "99% CI", "0.05": "95% CI", "0.10": "90% CI"}, 
                                   selected="0.05", inline=True),
        ),
        ui.input_action_button("rerun", "Run New Experiment", class_="btn-primary w-100"),
        ui.output_text_verbatim("stats_text"),
        ui.output_plot("ci_plot")
    )
)

# 2. Server Logic
def server(input, output, session):
    
    @render.plot
    def ci_plot():
        input.rerun() # Trigger redraw on button click
        
        # Get Inputs
        N = 100  
        n = input.sample_size()
        true_p = input.true_p()
        alpha = float(input.alpha())
        
        # Generate Data
        successes = np.random.binomial(n, true_p, N)
        sample_ps = successes / n
        
        # Calculate CIs (Using hardcoded Z-scores to avoid heavy Scipy import)
        z_map = {0.01: 2.576, 0.05: 1.96, 0.10: 1.645}
        z_score = z_map.get(alpha, 1.96)
        
        se = np.sqrt((sample_ps * (1 - sample_ps)) / n)
        lower = sample_ps - (z_score * se)
        upper = sample_ps + (z_score * se)
        
        # Check Hits
        hit = (lower <= true_p) & (upper >= true_p)
        colors = np.where(hit, 'lightgray', '#D9534F') # Red for miss
        widths = np.where(hit, 1, 2.5)
        
        # Plotting (Made shorter: figsize 8x3.5)
        fig, ax = plt.subplots(figsize=(8, 3.5), constrained_layout=True)
        
        ax.axvline(true_p, color='black', linestyle='--', linewidth=2, alpha=0.8)
        y_pos = np.arange(N)
        ax.hlines(y_pos, lower, upper, colors=colors, linewidth=widths)
        
        ax.set_yticks([])
        ax.set_xlim(0, 1)
        # Remove top/right spines for cleaner look
        ax.spines['top'].set_visible(False)
        ax.spines['right'].set_visible(False)
        ax.spines['left'].set_visible(False)
        
        return fig

    @render.text
    def stats_text():
        input.rerun()
        alpha = float(input.alpha())
        # Calculate z-score again for the text display
        z_map = {0.01: 2.576, 0.05: 1.96, 0.10: 1.645}
        z = z_map.get(alpha, 1.96)
        
        return (f"Simulation: 100 Experiments | Target Confidence: {100*(1-alpha):.0f}%\n"
                f"Interpretation: We expect about {int(100*alpha)} red intervals.")

app = App(app_ui, server)
